# MyBaby v0.1.0 (MVP) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deliver a working, self-hosted baby tracker with Sleep/Feeding/Diaper plugins, Forward-Auth, security hardening, and Baby Buddy data import — deployed as a single Docker container.

**Architecture:** FastAPI backend with SQLAlchemy 2.0 + SQLite WAL, plugin-based architecture with filesystem discovery. React TypeScript frontend with Catppuccin theming. Single-container deployment via multi-stage Docker build (node → python). Auth via Authelia Forward-Auth headers.

**Tech Stack:** Python 3.12, FastAPI 0.115+, SQLAlchemy 2.0, Alembic, Pydantic 2, structlog, slowapi, starlette-csrf, React 18, TypeScript, Tailwind CSS 3, Vite 6, Vitest, pytest, uvicorn

**Repo:** [xRiErOS/myPrivateBabyTracker](https://github.com/xRiErOS/myPrivateBabyTracker)

**Spec:** `docs/superpowers/specs/2026-04-19-mybaby-architecture-design.md`

**Design System:** `DESIGN.md` — Portiert vom Home Dashboard. Definiert Farbsystem (Catppuccin Latte/Macchiato), Typografie, Komponenten-Bibliothek (12 Komponenten), Interaktionsmuster, PWA/iOS-Optimierungen. Muss als erste Datei ins neue Repo kopiert werden und gilt als bindende Designreferenz.

**Frontend-Basis:** Die bestehenden React-Komponenten aus `~/Obsidian/tools/home-dashboard/src/pages/baby/` werden als Ausgangsbasis portiert (JSX → TSX, API-Pfade anpassen, Plugin-Architektur-konform machen). Kein Neuschreiben — das bewährte UI wird übernommen und evolutionär erweitert.

```python
# backend/app/core/database.py
from collections.abc import AsyncGenerator
from sqlalchemy import event, text
from sqlalchemy.ext.asyncio import AsyncSession, async_sessionmaker, create_async_engine
from sqlalchemy.orm import DeclarativeBase


class Base(DeclarativeBase):
    pass


_engine = None
_session_factory = None


def init_db(database_url: str) -> None:
    """Initialize database engine. Configures WAL mode for SQLite."""
    global _engine, _session_factory

    # Convert sqlite:// to sqlite+aiosqlite:// for async
    if database_url.startswith("sqlite:///"):
        async_url = database_url.replace("sqlite:///", "sqlite+aiosqlite:///", 1)
    else:
        async_url = database_url

    _engine = create_async_engine(async_url, echo=False)
    _session_factory = async_sessionmaker(_engine, expire_on_commit=False)

    # Enable WAL mode for SQLite
    if "sqlite" in database_url:

        @event.listens_for(_engine.sync_engine, "connect")
        def _set_sqlite_pragma(dbapi_conn, connection_record):
            cursor = dbapi_conn.cursor()
            cursor.execute("PRAGMA journal_mode=WAL")
            cursor.execute("PRAGMA foreign_keys=ON")
            cursor.close()


async def get_session() -> AsyncGenerator[AsyncSession, None]:
    """Yield a database session for FastAPI dependency injection."""
    if _session_factory is None:
        raise RuntimeError("Database not initialized. Call init_db() first.")
    async with _session_factory() as session:
        yield session


async def dispose_engine() -> None:
    """Dispose engine on shutdown. Runs WAL checkpoint for SQLite."""
    global _engine
    if _engine:
        # WAL checkpoint before shutdown
        if "sqlite" in str(_engine.url):
            async with _engine.begin() as conn:
                await conn.execute(text("PRAGMA wal_checkpoint(TRUNCATE)"))
        await _engine.dispose()
        _engine = None
```

```python
# backend/app/core/models.py
from datetime import datetime
from sqlalchemy import DateTime, ForeignKey, Integer, String, Text, UniqueConstraint, func
from sqlalchemy.orm import Mapped, mapped_column, relationship
from app.core.database import Base


class TimestampMixin:
    """Reusable mixin — all timestamps are UTC."""
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), nullable=False
    )


class User(TimestampMixin, Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    username: Mapped[str] = mapped_column(String(100), unique=True, nullable=False)
    email: Mapped[str | None] = mapped_column(String(255))
    password_hash: Mapped[str | None] = mapped_column(String(255))
    role: Mapped[str] = mapped_column(String(20), nullable=False, default="caregiver")
    display_name: Mapped[str | None] = mapped_column(String(200))
    locale: Mapped[str] = mapped_column(String(10), default="de")


class Child(TimestampMixin, Base):
    __tablename__ = "children"

    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    name: Mapped[str] = mapped_column(String(200), nullable=False)
    birth_date: Mapped[str] = mapped_column(String(10), nullable=False)  # ISO date
    gender: Mapped[str | None] = mapped_column(String(20))


class ChildCaregiver(Base):
    __tablename__ = "child_caregivers"

    child_id: Mapped[int] = mapped_column(
        Integer, ForeignKey("children.id", ondelete="CASCADE"), primary_key=True
    )
    user_id: Mapped[int] = mapped_column(
        Integer, ForeignKey("users.id", ondelete="CASCADE"), primary_key=True
    )


class PluginState(Base):
    __tablename__ = "plugin_states"
    __table_args__ = (UniqueConstraint("plugin_name", "child_id"),)

    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    plugin_name: Mapped[str] = mapped_column(String(50), nullable=False)
    child_id: Mapped[int | None] = mapped_column(Integer, ForeignKey("children.id"))
    is_active: Mapped[int] = mapped_column(Integer, nullable=False, default=0)
    activated_at: Mapped[datetime | None] = mapped_column(DateTime(timezone=True))


class PluginSettings(Base):
    __tablename__ = "plugin_settings"
    __table_args__ = (UniqueConstraint("plugin_name", "child_id"),)

    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    plugin_name: Mapped[str] = mapped_column(String(50), nullable=False)
    child_id: Mapped[int | None] = mapped_column(Integer, ForeignKey("children.id"))
    settings_json: Mapped[str] = mapped_column(Text, nullable=False, default="{}")
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), nullable=False
    )
```

```python
from datetime import datetime, timedelta, timezone
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import select

from app.core.database import get_session
from app.core.dependencies import get_current_user
from app.core.models import User

from app.plugins.sleep.schemas import SleepCreate, SleepUpdate, SleepResponse

@router.post("/", response_model=SleepResponse, status_code=201)
async def create_sleep(
    data: SleepCreate,
    db: AsyncSession = Depends(get_session),
    user: User = Depends(get_current_user),
):
    entry = SleepEntry(**data.model_dump(), user_id=user.id)
    db.add(entry)
    await db.commit()
    await db.refresh(entry)
    return entry


@router.get("/", response_model=list[SleepResponse])
async def list_sleep(
    child_id: int,
    date: str | None = None,
    limit: int = 100,
    db: AsyncSession = Depends(get_session),
    user: User = Depends(get_current_user),
):
    query = select(SleepEntry).where(SleepEntry.child_id == child_id)
    if date:
        day_start = datetime.fromisoformat(f"{date}T00:00:00+00:00")
        day_end = day_start + timedelta(days=1)
        query = query.where(SleepEntry.start >= day_start, SleepEntry.start < day_end)
    query = query.order_by(SleepEntry.start.desc()).limit(limit)
    result = await db.execute(query)
    return result.scalars().all()


@router.patch("/{entry_id}", response_model=SleepResponse)
async def update_sleep(
    entry_id: int,
    data: SleepUpdate,
    db: AsyncSession = Depends(get_session),
    user: User = Depends(get_current_user),
):
    entry = await db.get(SleepEntry, entry_id)
    if not entry:
        raise HTTPException(404, "Sleep entry not found")
    for field, value in data.model_dump(exclude_unset=True).items():
        setattr(entry, field, value)
    await db.commit()
    await db.refresh(entry)
    return entry


@router.delete("/{entry_id}", status_code=204)

    entry_id: int,
    db: AsyncSession = Depends(get_session),
    user: User = Depends(get_current_user),
):
    entry = await db.get(SleepEntry, entry_id)
    if not entry:
        raise HTTPException(404, "Sleep entry not found")
```

```python
# backend/app/core/auth.py
import ipaddress
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware
import structlog

logger = structlog.get_logger()


class ForwardAuthMiddleware(BaseHTTPMiddleware):
    """Reads Remote-User/Groups/Name headers from trusted proxies."""

    def __init__(self, app, trusted_header: str, trusted_proxies: str):
        super().__init__(app)
        self.trusted_header = trusted_header
        self.trusted_networks = [
            ipaddress.ip_network(cidr.strip()) for cidr in trusted_proxies.split(",")
        ]

    def _is_trusted(self, client_ip: str) -> bool:
        try:
            ip = ipaddress.ip_address(client_ip)
            return any(ip in net for net in self.trusted_networks)
        except ValueError:
            return False

    async def dispatch(self, request: Request, call_next):
        client_ip = request.client.host if request.client else "unknown"

        # K1: Strip ALL Remote-* headers first — always
        remote_user = None
        remote_groups = None
        remote_name = None

        if self._is_trusted(client_ip):
            remote_user = request.headers.get(self.trusted_header)
            remote_groups = request.headers.get("Remote-Groups", "")
            remote_name = request.headers.get("Remote-Name", "")
        else:
            # Log if someone tries to send Remote-* from untrusted IP
            if self.trusted_header in request.headers:
                logger.warning(
                    "forward_auth_header_from_untrusted_ip",
                    client_ip=client_ip,
                    header=self.trusted_header,
                )

        # Set on request.state for dependencies
        request.state.username = remote_user
        request.state.role = "admin" if "admin" in remote_groups else "caregiver"
        request.state.display_name = remote_name or remote_user

        return await call_next(request)
```

```python
import uuid
from fastapi import Request, Response
from starlette.middleware.base import BaseHTTPMiddleware
import structlog


class SecurityHeadersMiddleware(BaseHTTPMiddleware):
    """K2: Add CSP and security headers to all responses."""

    CSP = (
        "default-src 'self'; "
        "script-src 'self'; "
        "style-src 'self' 'unsafe-inline'; "
        "img-src 'self' data:; "
        "font-src 'self'; "
        "connect-src 'self'; "
        "frame-ancestors 'none'"
    )

    async def dispatch(self, request: Request, call_next):
        response: Response = await call_next(request)
        response.headers["Content-Security-Policy"] = self.CSP
        response.headers["X-Content-Type-Options"] = "nosniff"
        response.headers["X-Frame-Options"] = "DENY"
        response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
        return response


class RequestIdMiddleware(BaseHTTPMiddleware):
    """Add request ID for log correlation."""

    async def dispatch(self, request: Request, call_next):
        request_id = str(uuid.uuid4())[:8]
        structlog.contextvars.clear_contextvars()
        structlog.contextvars.bind_contextvars(request_id=request_id)
        response = await call_next(request)
        response.headers["X-Request-ID"] = request_id
        return response
```

```typescript
// frontend/src/core/api.ts
const API_BASE = '/api'

function getCsrfToken(): string | null {
  const match = document.cookie.match(/csrftoken=([^;]+)/)
  return match ? match[1] : null
}

export async function apiFetch<T>(path: string, options: RequestInit = {}): Promise<T> {
  const headers: Record<string, string> = {
    'Content-Type': 'application/json',
    ...(options.headers as Record<string, string> || {}),
  }

  // Add CSRF token for state-changing methods
  const method = (options.method || 'GET').toUpperCase()
  if (['POST', 'PATCH', 'PUT', 'DELETE'].includes(method)) {
    const csrf = getCsrfToken()
    if (csrf) headers['X-CSRF-Token'] = csrf
  }

  const resp = await fetch(`${API_BASE}${path}`, { ...options, headers })

  if (!resp.ok) {
    const body = await resp.text().catch(() => '')
    throw new Error(`API ${resp.status}: ${body || path}`)
  }

  if (resp.status === 204) return null as T
  return resp.json()
}
```

```typescript
interface AuthContextType {
  user: User | null
  loading: boolean
}

const AuthContext = createContext<AuthContextType>({ user: null, loading: true })

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    apiFetch<User>('/auth/me')
      .then(setUser)
      .catch(() => setUser(null))
      .finally(() => setLoading(false))
  }, [])

  return (
    <AuthContext.Provider value={{ user, loading }}>
      {children}
    </AuthContext.Provider>
  )
}

export function useAuth() {
  return useContext(AuthContext)
```

```python
# scripts/import_babybuddy.py
"""One-time migration: Baby Buddy REST API → MyBaby SQLite.

Usage:
  python scripts/import_babybuddy.py --api-url http://192.168.178.185:8090/api \
    --api-token 11131693473206b23199846e627d7c6384cd6571 \
    --db sqlite:///data/mybaby.db \
    --dry-run

Supports: feedings, changes (→ diaper), sleep, notes
"""
import argparse
import json
import sys
from datetime import datetime, timezone
import requests
from sqlalchemy import create_engine
from sqlalchemy.orm import Session

from app.core.database import Base
from app.core.models import User, Child
from app.plugins.sleep.models import SleepEntry
from app.plugins.feeding.models import FeedingEntry
from app.plugins.diaper.models import DiaperEntry


def fetch_all(api_url: str, token: str, endpoint: str) -> list[dict]:
    """Fetch all pages from Baby Buddy API."""
    headers = {"Authorization": f"Token {token}"}
    results = []
    url = f"{api_url}/{endpoint}/?format=json&limit=100"
    while url:
        resp = requests.get(url, headers=headers)
        resp.raise_for_status()
        data = resp.json()
        results.extend(data.get("results", data))
        url = data.get("next")
    return results


def import_data(api_url: str, token: str, db_url: str, dry_run: bool = True):
    engine = create_engine(db_url)
    Base.metadata.create_all(engine)

    feedings = fetch_all(api_url, token, "feedings")
    changes = fetch_all(api_url, token, "changes")
    sleeps = fetch_all(api_url, token, "sleep")
    children = fetch_all(api_url, token, "children")

    print(f"Found: {len(feedings)} feedings, {len(changes)} diapers, {len(sleeps)} sleeps, {len(children)} children")

    if dry_run:
        print("DRY RUN — no data written. Remove --dry-run to import.")
        return

    with Session(engine) as session:
        # Import children
        child_map = {}
        for c in children:
            child = Child(name=f"{c['first_name']} {c['last_name']}".strip(), birth_date=c['birth_date'])
            session.add(child)
            session.flush()
            child_map[c['id']] = child.id

        # Create import user
        import_user = User(username="import", role="admin", display_name="Baby Buddy Import")
        session.add(import_user)
        session.flush()

        # Import feedings
        for f in feedings:
            session.add(FeedingEntry(
                child_id=child_map.get(f['child'], 1),
                user_id=import_user.id,
                timestamp=f['start'],
                type="bottle" if f.get('type') == 'bottle' else "breast",
                amount_ml=f.get('amount'),
                duration_min=f.get('duration') // 60 if f.get('duration') else None,
                notes=f.get('notes', ''),
            ))

        # Import changes → diaper
        for c in changes:
            session.add(DiaperEntry(
                child_id=child_map.get(c['child'], 1),
                user_id=import_user.id,
                timestamp=c['time'],
                wet=1 if c.get('wet') else 0,
                solid=1 if c.get('solid') else 0,
                color=c.get('color', '').lower() if c.get('color') else None,
            ))

        # Import sleep
        for s in sleeps:
            session.add(SleepEntry(
                child_id=child_map.get(s['child'], 1),
                user_id=import_user.id,
                start=s['start'],
                end=s.get('end'),
                type="nap" if s.get('nap') else "night",
            ))

        session.commit()
        print(f"Imported: {len(feedings)} feedings, {len(changes)} diapers, {len(sleeps)} sleeps")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Import data from Baby Buddy")
    parser.add_argument("--api-url", required=True)
    parser.add_argument("--api-token", required=True)
    parser.add_argument("--db", default="sqlite:///data/mybaby.db")
    parser.add_argument("--dry-run", action="store_true")
    args = parser.parse_args()
    import_data(args.api_url, args.api_token, args.db, args.dry_run)
```
